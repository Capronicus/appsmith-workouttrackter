SELECT
    m.name,
    m.body_region,
    SUM(em.intensity) AS intensity
FROM template_exercises te
JOIN exercise_muscles em
    ON em.exercise_id = te.exercise_id
JOIN muscles m
    ON m.id = em.muscle_id
WHERE te.template_id = {{ SelectWorkout.selectedOptionValue }}
GROUP BY
    m.name,
    m.body_region
ORDER BY intensity DESC;
