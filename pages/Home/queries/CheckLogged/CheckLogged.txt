SELECT s.id AS session_id, s.session_date, s.completed_at,
       wt.name AS workout_name,
       e.name AS exercise_name,
       ss.set_number, ss.reps, ss.weight, ss.feel, ss.exercise_notes
FROM sessions s
LEFT JOIN workout_templates wt ON wt.id = s.template_id
LEFT JOIN session_sets ss ON ss.session_id = s.id
LEFT JOIN exercises e ON e.id = ss.exercise_id
ORDER BY s.id DESC, ss.exercise_id, ss.set_number
LIMIT 50;
